def diff(big: Z, small: Z): Z = {
    Contract(
        Requires( big >= small ),
        Ensures ( result == big-small )
    )

    //PREMISES:



    var count: Z = 0
    var cur: Z = big

    //PREMISES:



    //SHOW:



    while (cur != small) {
        Invariant(
            Modifies(count, cur),
            count == (big-cur)
        )

        //PREMISES:



        cur = cur - 1
        count = count + 1

        //SHOW:


    }

    //PREMISES:



    //SHOW:



    return count
}
